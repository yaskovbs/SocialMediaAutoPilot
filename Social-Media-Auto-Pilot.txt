        return base44.entities.ChannelSettings.create(data);

Parse all videos from the feed. For each video return:

- published_at: the publication date in ISO 8601 format
- youtube_url: full YouTube URL

- description: short description (first 200 chars)
- video_type: "short" if it's a YouTube Short, "video" otherwise

Also return the channel_name.
Return videos sorted from OLDEST to NEWEST (by published_at date).`,
      add_context_from_internet: true,

                title: { type: "string" },
                published_at: { type: "string" },
                youtube_url: { type: "string" },
                thumbnail_url: { type: "string" },
                description: { type: "string" },
                video_type: { type: "string" }
              }
            }
          }
        }
      }

      });

          </div>
          <div>
            <h2 className="text-white font-semibold">ערוץ YouTube</h2>
            <p className="text-gray-400 text-xs">{localSettings.youtube_channel_name || 'לא מחובר'}</p>
          </div>

        </div>

import React, { useState } from 'react';
import { base44 } from '@/api/base44Client';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { Search, Filter } from 'lucide-react';
import VideoCard from '../components/VideoCard';

const PLATFORMS = ['tiktok', 'instagram', 'facebook', 'x'];

export default function Videos() {
  const [typeFilter, setTypeFilter] = useState('all'); // all, video, short
  const [statusFilter, setStatusFilter] = useState('all'); // all, pending, posted, failed
  const [searchQuery, setSearchQuery] = useState('');
  
  const queryClient = useQueryClient();

  const { data: videos = [], isLoading } = useQuery({
    queryKey: ['videos'],
    queryFn: () => base44.entities.Video.list('published_at', 200),
  });

  const updateStatus = useMutation({
    mutationFn: ({ id, platform, status }) => 
      base44.entities.Video.update(id, { [`${platform}_status`]: status }),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ['videos'] }),
  });

  const filteredVideos = videos.filter(v => {
    if (typeFilter !== 'all' && v.video_type !== typeFilter) return false;
    if (statusFilter === 'pending' && !PLATFORMS.some(p => v[`${p}_status`] === 'pending')) return false;
    if (statusFilter === 'posted' && !PLATFORMS.every(p => v[`${p}_status`] === 'posted')) return false;
    if (statusFilter === 'failed' && !PLATFORMS.some(p => v[`${p}_status`] === 'failed')) return false;
    if (searchQuery && !v.title?.toLowerCase().includes(searchQuery.toLowerCase())) return false;
    return true;
  });

  const typeFilters = [
    { id: 'all', label: 'הכל' },
    { id: 'video', label: 'סרטונים' },
    { id: 'short', label: 'Shorts' },
  ];

  const statusFilters = [
    { id: 'all', label: 'כל הסטטוסים' },
    { id: 'pending', label: 'ממתינים' },
    { id: 'posted', label: 'פורסם' },
    { id: 'failed', label: 'נכשל' },
  ];

  return (
    <div className="p-6 max-w-7xl mx-auto space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold text-white">סרטונים</h1>
          <p className="text-gray-400 text-sm mt-1">{filteredVideos.length} סרטונים</p>
        </div>
      </div>

      {/* Filters */}
      <div className="flex flex-wrap gap-3 items-center">
        <div className="relative">
          <Search className="absolute right-3 top-1/2 -translate-y-1/2 w-4 h-4 text-gray-500" />
          <input
            type="text"
            placeholder="חיפוש..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className="bg-gray-900 border border-gray-700 rounded-xl px-4 py-2 pr-9 text-white text-sm placeholder-gray-500 focus:outline-none focus:border-red-500 w-48"
          />
        </div>
        
        <div className="flex gap-1 bg-gray-900 rounded-xl p-1 border border-gray-800">
          {typeFilters.map(f => (
            <button
              key={f.id}
              onClick={() => setTypeFilter(f.id)}
              className={`px-3 py-1.5 rounded-lg text-sm font-medium transition-colors ${
                typeFilter === f.id ? 'bg-red-600 text-white' : 'text-gray-400 hover:text-gray-200'
              }`}
            >
              {f.label}
            </button>
          ))}
        </div>

        <div className="flex gap-1 bg-gray-900 rounded-xl p-1 border border-gray-800">

// pages/Settings.jsx
import React, { useState, useEffect } from 'react';
import { base44 } from '@/api/base44Client';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { RefreshCw, Check, AlertCircle, Youtube } from 'lucide-react';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Switch } from '@/components/ui/switch';
import { Label } from '@/components/ui/label';

const PLATFORMS = [
  { id: 'tiktok', label: 'TikTok', color: 'from-gray-900 to-gray-800', dot: 'bg-gray-500' },
  { id: 'instagram', label: 'Instagram', color: 'from-pink-950 to-gray-900', dot: 'bg-pink-500' },
  { id: 'facebook', label: 'Facebook', color: 'from-blue-950 to-gray-900', dot: 'bg-blue-500' },
  { id: 'x', label: 'X (Twitter)', color: 'from-gray-900 to-gray-800', dot: 'bg-gray-300' },
];

export default function Settings() {
  const queryClient = useQueryClient();
  const [channelId, setChannelId] = useState('');
  const [isSyncing, setIsSyncing] = useState(false);
  const [syncResult, setSyncResult] = useState(null);
  const [localSettings, setLocalSettings] = useState({});

  const { data: settingsList = [] } = useQuery({
    queryKey: ['channelSettings'],
    queryFn: () => base44.entities.ChannelSettings.list(),
  });

  const settings = settingsList[0];

  useEffect(() => {
    if (settings) {
      setChannelId(settings.youtube_channel_id || '');
      setLocalSettings(settings);
    }
  }, [settings]);

  const saveSettings = useMutation({
    mutationFn: async (data) => {
      if (settings?.id) {
        return base44.entities.ChannelSettings.update(settings.id, data);
      } else {
        return base44.entities.ChannelSettings.create(data);
      }
    },
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ['channelSettings'] }),
  });

  const handleFieldChange = (field, value) => {
    const updated = { ...localSettings, [field]: value };
    setLocalSettings(updated);
    saveSettings.mutate(updated);
  };

  const syncYouTube = async () => {
    if (!channelId.trim()) return;
    
    setIsSyncing(true);
    setSyncResult(null);
    
    // Extract channel ID from URL if needed
    let cleanId = channelId.trim();
    if (cleanId.includes('youtube.com')) {
      const match = cleanId.match(/channel\/([^/?]+)/);
      if (match) cleanId = match[1];
    }
    
    const result = await base44.integrations.Core.InvokeLLM({
      prompt: `Fetch all available YouTube videos from the channel with ID: ${cleanId}
      
Please fetch the RSS feed at: https://www.youtube.com/feeds/videos.xml?channel_id=${cleanId}

Parse all videos from the feed. For each video return:
- youtube_id: the video ID (just the ID, not the full URL)
- title: the video title
- published_at: the publication date in ISO 8601 format
- youtube_url: full YouTube URL
- thumbnail_url: use format https://img.youtube.com/vi/{VIDEO_ID}/mqdefault.jpg
- description: short description (first 200 chars)
- video_type: "short" if it's a YouTube Short, "video" otherwise

Also return the channel_name.
Return videos sorted from OLDEST to NEWEST (by published_at date).`,
      add_context_from_internet: true,
      response_json_schema: {
        type: "object",
        properties: {
          channel_name: { type: "string" },
          videos: {
            type: "array",
            items: {
              type: "object",
              properties: {
                youtube_id: { type: "string" },
                title: { type: "string" },
                published_at: { type: "string" },
                youtube_url: { type: "string" },
                thumbnail_url: { type: "string" },
                description: { type: "string" },
                video_type: { type: "string" }
              }
            }
          }
        }
      }
    });
    
    if (result?.videos?.length > 0) {
      // Get existing videos to avoid duplicates
      const existingVideos = await base44.entities.Video.list();
      const existingIds = new Set(existingVideos.map(v => v.youtube_id));
      
      const newVideos = result.videos
        .filter(v => v.youtube_id && !existingIds.has(v.youtube_id))
        .map(v => ({
          ...v,
          video_type: v.video_type || 'video',
          tiktok_status: 'pending',
          instagram_status: 'pending',
          facebook_status: 'pending',
          x_status: 'pending',
        }));
      
      if (newVideos.length > 0) {
        await base44.entities.Video.bulkCreate(newVideos);
      }
      
      // Save channel settings
      await saveSettings.mutateAsync({
        ...localSettings,
        youtube_channel_id: cleanId,
        youtube_channel_name: result.channel_name || cleanId,
        last_sync: new Date().toISOString(),
        total_synced: (settings?.total_synced || 0) + newVideos.length,
      });
      
      setSyncResult({ success: true, count: newVideos.length, total: result.videos.length });
      queryClient.invalidateQueries({ queryKey: ['videos'] });
      queryClient.invalidateQueries({ queryKey: ['channelSettings'] });
    } else {
      setSyncResult({ success: false, message: 'לא נמצאו סרטונים. בדוק את ה-Channel ID' });
    }
    
    setIsSyncing(false);
  };

  return (
    <div className="p-6 max-w-3xl mx-auto space-y-6">
      <div>
        <h1 className="text-2xl font-bold text-white">הגדרות</h1>
        <p className="text-gray-400 text-sm mt-1">חיבור ערוץ YouTube ופלטפורמות</p>
      </div>

      {/* YouTube Section */}
      <div className="bg-gray-900 rounded-2xl p-6 border border-gray-800 space-y-4">
        <div className="flex items-center gap-3 mb-5">
          <div className="w-10 h-10 bg-red-600/20 rounded-xl flex items-center justify-center">
            <Youtube className="w-5 h-5 text-red-500" />
          </div>
          <div>
            <h2 className="text-white font-semibold">ערוץ YouTube</h2>
            <p className="text-gray-400 text-xs">{localSettings.youtube_channel_name || 'לא מחובר'}</p>
          </div>
        </div>

        <div>
          <Label className="text-gray-400 text-sm mb-1.5 block">Channel ID</Label>
          <div className="flex gap-2">
            <Input
              value={channelId}
              onChange={(e) => setChannelId(e.target.value)}
              placeholder="UC... (הזן Channel ID של הערוץ)"
              className="bg-gray-800 border-gray-700 text-white placeholder-gray-600 flex-1"
              dir="ltr"
            />
            <Button
              onClick={syncYouTube}
              disabled={isSyncing || !channelId.trim()}
              className="bg-red-600 hover:bg-red-700 gap-2"
            >
              <RefreshCw className={`w-4 h-4 ${isSyncing ? 'animate-spin' : ''}`} />
              {isSyncing ? 'מסנכרן...' : 'סנכרן'}
            </Button>
          </div>
          <p className="text-gray-600 text-xs mt-2">
            למצוא Channel ID: youtube.com → הערוץ שלך → פרטי ערוץ → Channel ID
          </p>
        </div>

        {syncResult && (
          <div className={`flex items-center gap-2 p-3 rounded-xl text-sm ${
            syncResult.success 
              ? 'bg-green-900/20 border border-green-800/50 text-green-400' 
              : 'bg-red-900/20 border border-red-800/50 text-red-400'
          }`}>
            {syncResult.success ? <Check className="w-4 h-4" /> : <AlertCircle className="w-4 h-4" />}
            {syncResult.success 
              ? `נמצאו ${syncResult.total} סרטונים, ${syncResult.count} חדשים נוספו`
              : syncResult.message
            }
          </div>
        )}

        {localSettings.last_sync && (
          <p className="text-gray-600 text-xs">
            סנכרון אחרון: {new Date(localSettings.last_sync).toLocaleString('he-IL')}
          </p>
        )}
      </div>

      {/* Platform Connections */}
      <div className="bg-gray-900 rounded-2xl p-6 border border-gray-800 space-y-4">
        <h2 className="text-white font-semibold mb-5">חיבור פלטפורמות</h2>
        
        <div className="bg-amber-900/20 border border-amber-800/50 rounded-xl p-3 text-amber-400 text-xs flex gap-2">
          <AlertCircle className="w-4 h-4 flex-shrink-0 mt-0.5" />
          <span>פרסום אוטומטי מלא דורש הפעלת Backend Functions. כעת ניתן לנהל את הסטטוס ידנית.</span>
        </div>

        {PLATFORMS.map((platform) => {
          const isConnected = localSettings[`${platform.id}_connected`];
          const username = localSettings[`${platform.id}_username`] || localSettings[`${platform.id}_page_name`];
          
          return (
            <div key={platform.id} className={`rounded-xl p-4 border border-gray-700 bg-gradient-to-l ${platform.color}`}>
              <div className="flex items-center justify-between mb-3">
                <div className="flex items-center gap-2">
                  <div className={`w-2.5 h-2.5 rounded-full ${platform.dot}`} />
                  <span className="text-white font-medium">{platform.label}</span>
                </div>
                <Switch
                  checked={isConnected || false}
                  onCheckedChange={(checked) => handleFieldChange(`${platform.id}_connected`, checked)}
                />
              </div>
              {isConnected && (
                <Input
                  value={username || ''}
                  onChange={(e) => handleFieldChange(`${platform.id}_username`, e.target.value)}
                  placeholder={`@שם_משתמש`}
                  className="bg-gray-800/50 border-gray-600 text-white placeholder-gray-600 text-sm h-8"
                />
              )}
            </div>
          );
        })}
      </div>

      {/* Post Settings */}
      <div className="bg-gray-900 rounded-2xl p-6 border border-gray-800">
        <h2 className="text-white font-semibold mb-4">הגדרות פרסום</h2>
        <div className="space-y-4">
          <div className="flex items-center justify-between">
            <div>
              <p className="text-white text-sm font-medium">כלול Shorts</p>
              <p className="text-gray-500 text-xs">פרסם גם סרטונים קצרים</p>
            </div>
            <Switch
              checked={localSettings.include_shorts !== false}
              onCheckedChange={(v) => handleFieldChange('include_shorts', v)}
            />
          </div>
          <div className="flex items-center justify-between">
            <div>
              <p className="text-white text-sm font-medium">סדר פרסום</p>
              <p className="text-gray-500 text-xs">מהישן לחדש</p>
            </div>
            <span className="text-green-400 text-xs bg-green-900/20 px-2 py-1 rounded-lg">פעיל</span>
          </div>
        </div>
      </div>
    </div>
  );
}
